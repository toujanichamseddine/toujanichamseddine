    <script>
        // Function to generate random data for demo purposes
        function getRandomValue(min, max) {
            return Math.floor(Math.random() * (max - min + 1) + min);
        }
        
        // Update stats periodically
        function updateStats() {
            // Update server status
            document.getElementById('uptime').textContent = getRandomValue(1, 30) + 'd ' + 
                                                          getRandomValue(0, 23) + 'h ' + 
                                                          getRandomValue(0, 59) + 'm';
            document.getElementById('loadAvg').textContent = (Math.random() * 3).toFixed(2);
            document.getElementById('processes').textContent = getRandomValue(100, 200);
            document.getElementById('temperature').textContent = getRandomValue(40, 75) + '°C';
            
            // Update resource usage
            const cpuUsage = getRandomValue(10, 90);
            document.getElementById('cpuUsage').textContent = cpuUsage + '%';
            document.getElementById('cpuProgress').style.width = cpuUsage + '%';
            document.getElementById('cpuProgress').setAttribute('aria-valuenow', cpuUsage);
            
            const memoryUsage = getRandomValue(30, 90);
            document.getElementById('memoryUsage').textContent = memoryUsage + '%';
            document.getElementById('memoryProgress').style.width = memoryUsage + '%';
            document.getElementById('memoryProgress').setAttribute('aria-valuenow', memoryUsage);
            
            const diskUsage = getRandomValue(20, 80);
            document.getElementById('diskUsage').textContent = diskUsage + '%';
            document.getElementById('diskProgress').style.width = diskUsage + '%';
            document.getElementById('diskProgress').setAttribute('aria-valuenow', diskUsage);
            
            // Update network stats
            document.getElementById('inboundTraffic').textContent = (Math.random() * 5).toFixed(2) + ' MB/s';
            document.getElementById('outboundTraffic').textContent = (Math.random() * 5).toFixed(2) + ' MB/s';
            
            // Update last updated time
            const now = new Date();
            document.getElementById('lastUpdated').textContent = now.toLocaleTimeString();
            
            // Update charts
            updateCharts();
        }
