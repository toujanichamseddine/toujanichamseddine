/**
     * Vérifie si l'utilisateur possède un rôle spécifique
     * @param role Le rôle à vérifier
     * @return true si l'utilisateur possède le rôle, false sinon
     */
    public boolean hasRole(String role) {
        if (roles == null || role == null) {
            return false;
        }
        
        // Vérifier avec et sans le préfixe ROLE_
        String roleToCheck = role.startsWith("ROLE_") ? role : "ROLE_" + role;
        String simplifiedRole = role.startsWith("ROLE_") ? role.substring(5) : role;
        
        return roles.contains(roleToCheck) || roles.contains(simplifiedRole);
    }

    /**
     * Vérifie si l'utilisateur possède au moins un des rôles spécifiés
     * @param rolesToCheck Les rôles à vérifier
     * @return true si l'utilisateur possède au moins un des rôles, false sinon
     */
    public boolean hasAnyRole(String... rolesToCheck) {
        if (roles == null || rolesToCheck == null) {
            return false;
        }
        
        for (String role : rolesToCheck) {
            if (hasRole(role)) {
                return true;
            }
        }
        
        return false;
    }
