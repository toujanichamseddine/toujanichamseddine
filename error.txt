import java.nio.charset.StandardCharsets;
import java.util.Base64;
import java.util.zip.Inflater;

public class SamlDecoder {
    public static void main(String[] args) throws Exception {
        String samlResponse = "YOUR_BASE64_STRING";

        // Step 1: Base64 decode
        byte[] decoded = Base64.getDecoder().decode(samlResponse);

        // Step 2: Inflate (raw DEFLATE stream, 'nowrap=true')
        Inflater inflater = new Inflater(true);
        inflater.setInput(decoded);

        byte[] buffer = new byte[10240];
        int len = inflater.inflate(buffer);
        inflater.end();

        String xml = new String(buffer, 0, len, StandardCharsets.UTF_8);
        System.out.println(xml);
    }
}
